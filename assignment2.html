Firmware

                        const int buttonPin = 3;     // the number of the pushbutton pin
                        const int ledPin =  13;      // the number of the LED pin
                        int buttonState = 0;         // variable for reading the pushbutton status
                        
                        void setup() {
                          pinMode(ledPin, OUTPUT); //sets pin 13 as output
                          pinMode(buttonPin, INPUT); //sets pin 2 as input
                        }
                        
                        void loop() {
                        buttonState = digitalRead(buttonPin); //reads  stat of button
                        
                          // check if the pushbutton is pressed. If it is, the buttonState is HIGH:
                          if (buttonState == HIGH) {
                            for(int fade = 255; fade >= 0; fade -= 5){ // sets increments for brightness of led
                              analogWrite(ledPin, fade); //led fades out at increments of 5
                              delay(30); //delay .03 seconds
                            }
                            
                          } else {  //if button is not pushed
                            digitalWrite(ledPin, LOW); //turns LED off
                          }
                        }
